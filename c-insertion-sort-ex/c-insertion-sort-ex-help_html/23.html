17     void insertion_sort(int *arr, int n){
18     
19     	//A simple function for sorting an array using insertion sort
20     
21     	   int i,j,key;
22     	   for (j = 1; j < n; j++)
23     	   {
24     	       key = arr[j];
25     	       i = j-1;
26     
27     	       /* Elements to the left of key (key = arr[j]),i.e.,elements of arr[0..j-1],
28     	          that are greater than key (i.e., arr[j]), move one position
29                   to the right, and arr[j] moves into the evacuated position*/
30     
31     	       while (i >= 0 && arr[i] > key)
32     	       {
33     	           arr[i+1] = arr[i];
34     	           i = i-1;
35     	       }
36     	       arr[i+1] = key;
37     	   }
38     }
